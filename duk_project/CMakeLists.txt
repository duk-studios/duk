
set(DUK_PROJECT_HEADERS
        include/duk_project/resource.h
        include/duk_project/register_types.h
        include/duk_project/project.h
        )
set(DUK_PROJECT_SOURCES
        src/duk_project/resource.cpp
        src/duk_project/register_types.cpp
        src/duk_project/project.cpp
        )

add_library(duk.project ${DUK_PROJECT_HEADERS} ${DUK_PROJECT_SOURCES})
add_library(duk::project ALIAS duk.project)

target_include_directories(duk.project
        PUBLIC
        $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>

        PRIVATE
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/src>
        )

target_link_libraries(duk.project
        PUBLIC
        duk::engine
        duk::animation
        )

set_target_properties(duk.project PROPERTIES EXPORT_NAME project)

install(
        DIRECTORY include/
        DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
)

install(
        TARGETS duk.project
        EXPORT duk-targets
)
